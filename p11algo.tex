\section*{Bubble Sort Algorithm}

Algorithm to sort an array of numbers using Bubble Sort.

\vspace{1cm}
\begin{algorithm}[H]
\caption{BubbleSort($A[1..n]$)}
\SetAlgoLined
\DontPrintSemicolon
\KwIn{An array $A$ of $n$ elements}
\KwOut{Array $A$ sorted in non-decreasing order}
\For{$i \gets 1$ \KwTo $n-1$}{
    \For{$j \gets 1$ \KwTo $n-i$}{
        \If{$A[j] > A[j+1]$}{
            swap $A[j]$ and $A[j+1]$\;
        }
    }
}
\Return{$A$}\;
\end{algorithm}
